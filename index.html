/* render関数内の枠表示ロジックを、さらに厳格に修正しました */ function render() { ctx.clearRect(0,0,cvs.width,cvs.height); ctx.save(); ctx.translate(cam.x, cam.y); ctx.scale(cam.z, cam.z); // ... (グリッドと線の描画はそのまま) ... map.sort((a,b)=>(Math.round(a.x)+Math.round(a.y))-(Math.round(b.x)+Math.round(b.y))).forEach((d, i) => { const a = assets[d.aIdx]; if(!a) return; const w = T_W * d.s, h = (a.img.height/a.img.width) * w, anchor = toS(d.x + a.gw + d.ox, d.y + a.gh + d.oy); ctx.drawImage(a.img, anchor.x - w / 2, anchor.y - h + d.off, w, h); // --- 【ここを修正】枠を表示する条件をより厳しく --- let showFrame = false; if (lockedId === d.id) { // 1. 今まさに指で触って動かしている（設置中含む） showFrame = true; } else if (activeAsset === -1 && selIdx === i) { // 2. 「何も設置しようとしていない時」かつ「選んでいる」 // つまり、倉庫で何も選んでいない時だけ、古い建物の枠が出る showFrame = true; } if(showFrame) { ctx.strokeStyle = "#007aff"; ctx.lineWidth = (lockedId ? 1 : 3)/cam.z; const gx = Math.round(d.x), gy = Math.round(d.y), pS = toS(gx+a.gw, gy+a.gh), pW = toS(gx, gy+a.gh), pE = toS(gx+a.gw, gy), pN = toS(gx, gy); ctx.beginPath(); ctx.moveTo(pS.x,pS.y); ctx.lineTo(pW.x,pW.y); ctx.lineTo(pN.x,pN.y); ctx.lineTo(pE.x,pE.y); ctx.closePath(); ctx.stroke(); } }); ctx.restore(); }
